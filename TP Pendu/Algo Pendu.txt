//BUT:L'utilisateur doit deviné le mot caché en cherchant lettre par lettre et en faisant le moins de fautes possibles
//ENTREE:1 caractere
//SORTIE:joueur gagne ou non

CONST
	MOT="rachid"
	

//BUT:Affiche un texte à l'utilisateur
//ENTREE:
//SORTIE:texte afficher
PROCEDURE afficherPresentation()
DEBUT
	//PARTIE PRESENTATION DU JEU A L'UTILISATEUR
	ECRIRE "Bonjour et bienvenue dans le jeu du pendu !"
	ECRIRE "Le but est de trouver le mot cacher en indiquant le caractere"
	ECRIRE "et en se trompant un minimum de fois,"
	ECRIRE "en sachant que le nombre d''erreur maximum est 3"
FINPROCEDURE


//BUT:Initialise la valeur d'une chaine
//ENTREE:1 chaine 1 entier
//SORTIE:1 chaine
PROCEDURE initmotTrouver(VAR motTrouver:CHAINE, tailleMot:ENTIER)
VAR
	i:ENTIER
DEBUT
	i<=0

	POUR i<=1 A tailleMot-1 FAIRE
		motTrouver[i]<='_';
	FINPOUR
FINPROCEDURE


//BUT:Verifie si la lettre existe ou non et remplace la variable
//ENTREE:2 entier, 1 caractere 1 chaine
//SORTIE:
PROCEDURE remplacemotTrouver(tailleMot:ENTIER,VAR erreur:ENTIER, lettre:CARACTERE, VAR motTrouver:CHAINE)
VAR
	i:ENTIER
	test:BOOLEEN
DEBUT
	i<=0
	test<=FAUX
	
	POUR i<=0 A tailleMot-1 FAIRE
		SI (lettre=MOT[i]) ALORS
			motTrouver[i]<=lettre
			ECRIRE "Trouver !"
			test<=VRAI
		FINSI
	FINPOUR
	
	SI (test)
		erreur<=erreur-1
		
		ECRIRE "Le caractere " & lettre & " n'est pas dans le mot"
		
		SI (erreur>0)
			ECRIRE "Il vous reste " & erreur & " chance(s) de trouver le mot"
			
			SINON
				ECRIRE "C'etais votre derniere chance !"
			FINSI
		FINSI
	FINSI
FINPROCEDURE


//BUT:verifie si un caractere existe dans une chaine
//ENTREE:1 caractere 1 chaine
//SORTIE:1 booleen
FONCTION testLettre(lettre:CARACTERE,lettreTester:CHAINE)
VAR
	i:ENTIER
	test:BOOLEEN
DEBUT
	test<=FAUX
	
	POUR i<=0 A tailleMot-1 FAIRE
		SI (lettreTester[i]=lettre)
			test<=VRAI
		FINSI
	FINPOUR

	testLettre<=test
FINFONCTION


//DEBUT PROGRAMME PRINCIPAL

VAR
	tailleMot,i,erreur,cpt:ENTIER
	motTrouver,lettreTester:CHAINE
	lettre:CARACTERE
	test:BOOLEEN
DEBUT
	
	//init du tableau de caractere motTrouver
	initmotTrouver(motTrouver,tailleMot)
	
	//PARTIE PRESENTATION DU JEU A L'UTILISATEUR
	afficherPresentation();

	ECRIRE "Le mot fait " & tailleMot & " caractere :"

	//début de boucle de jeu
	REPETER
		REPETER
			//demande à l'utilisateur d'entrer une lettre
			ECRIRE "Entrez un caractere: "
			LIRE lettre

			//test de la lettre si elle a deja ete tapper
			test<=testLettre(lettre,lettreTester);

			SI(test) ALORS/     //si le caractere à deja ete tapper
				ECRIRE Le caractere " & lettre & " a deja ete tapper !\n"
				erreur--;
				ECRIRE "Il vous rester " & erreur & " chance(s)\n"
			FINSI
		JUSQUA ((test) ET (erreur))

		SI (erreur>0) ALORS
			lettreTester[cpt]<=lettre
			cpt<=cpt+1
			
			//verification de la lettre avec le mot + nbr erreur
			remplacemotTrouver(tailleMot,erreur,lettre,motTrouver)
			
			//affiche le mot trouver (ex: R _ C _ _ _ )
			POUR i<=1 A tailleMot FAIRE
				ECRIRE motTrouver[i]
			FINPOUR
		FINSI
	
	JUSQUA ((motTrouver=MOT) OU (erreur>0))
	//fin de boucle de jeu

	//affiche a l'utilisateur si il a gagner ou perdu
	SI (erreur=FAUX)
		ECRIRE "Le mot etais " & MOT & ", vous ferez mieux la prochaine fois !"
		SINON
			ECRIRE "Le mot est bien " & erreur & ", bravo !\n",MOT
			ECRIRE "Il vous rester " & erreur & " chance(s)"
		FINSI
	FINSI

FIN.
